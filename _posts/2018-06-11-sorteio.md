## Código do Sorteio com uso do Tkinter:

```
from tkinter import *
from random import shuffle

def sort(members):

    file = open("resultado.txt", "w")

    file.write("Tempo total da rodada: %d min\n" %(len(members)*10))

    shuffle(members)

    for i in range(1,len(members)):
        file.write("\nDupla %d:\n Piloto: %s\n Co-piloto: %s\n" %(i,members[i-1],members[i]))
        if (i==len(members)-1) :
            file.write("\nDupla %d:\n Piloto: %s\n Co-piloto: %s\n" %(len(members),members[len(members)-1],members[0]))

    file.close()

def sortear():
   membros = [membro1.get(), membro2.get(), membro3.get(), membro4.get(), membro5.get()]
   sort(membros)

master = Tk()
Label(master, text="Membro 1: ").grid(row=0)
Label(master, text="Membro 2: ").grid(row=1)
Label(master, text="Membro 3: ").grid(row=2)
Label(master, text="Membro 4: ").grid(row=3)
Label(master, text="Membro 5: ").grid(row=4)

membro1 = Entry(master)
membro2 = Entry(master)
membro3 = Entry(master)
membro4 = Entry(master)
membro5 = Entry(master)

membro1.grid(row=0, column=1)
membro2.grid(row=1, column=1)
membro3.grid(row=2, column=1)
membro4.grid(row=3, column=1)
membro5.grid(row=4, column=1)

Button(master, text='Sair', command=master.quit).grid(row=5, column=0, sticky=W, pady=4)
Button(master, text='Fazer sorteio', command=sortear).grid(row=5, column=1, sticky=W, pady=4)

mainloop()
```
